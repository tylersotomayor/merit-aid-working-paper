% Identification Funnel: How Each Test Narrows the Set of Explanations
% Columbia University colors: Columbia Blue (#B9D9EB), dark blue (#1D4F91)

\definecolor{columbiablue}{HTML}{B9D9EB}
\definecolor{columbiadark}{HTML}{1D4F91}

\begin{tikzpicture}[
    every node/.style={font=\small},
    block/.style={draw=columbiadark, rounded corners, fill=#1,
                  minimum height=0.85cm, font=\small\bfseries,
                  align=center},
    elim/.style={font=\scriptsize, text=gray!50!black, align=center},
    arrow/.style={->, thick, columbiadark!40},
]

% Top: all explanations
\node[block=gray!20, text=black, minimum width=14cm] (top) at (0, 0)
    {All possible explanations for rising Q1 share at TN public colleges};

% Filter 1
\node[block=columbiadark!90, text=white, minimum width=13cm] (f1) at (0, -2.0)
    {(1) Level decomposition: real headcount increase, total enrollment flat};
\node[elim] at (0, -1.0)
    {\textit{Eliminates:} denominator artifacts, reporting changes, cohort size shifts};
\draw[arrow] (top.south) -- (f1.north);

% Filter 2
\node[block=columbiadark!75, text=white, minimum width=12cm] (f2) at (0, -4.0)
    {(2) Gender: female effect 2$\times$ male, matches GPA eligibility gap};
\node[elim] at (0, -3.0)
    {\textit{Eliminates:} gender-neutral confounders (state economy, demographics, federal policy)};
\draw[arrow] (f1.south) -- (f2.north);

% Filter 3
\node[block=columbiadark!60, text=white, minimum width=11cm] (f3) at (0, -6.0)
    {(3) Sector: effect at publics, null at privates in same state};
\node[elim] at (0, -5.0)
    {\textit{Eliminates:} state-wide shocks (fiscal, institutional, labor market)};
\draw[arrow] (f2.south) -- (f3.north);

% Filter 4
\node[block=columbiadark!45, text=columbiadark!90, minimum width=10cm] (f4) at (0, -8.0)
    {(4) Cross-state: MA replicates, SD inverts as model predicts};
\node[elim] at (0, -7.0)
    {\textit{Eliminates:} TN-specific confounders (any unobserved TN change circa 1993)};
\draw[arrow] (f3.south) -- (f4.north);

% Filter 5
\node[block=columbiablue, text=columbiadark, minimum width=9cm] (f5) at (0, -10.0)
    {(5) Causal forest: heterogeneity aligns with theory-motivated covariates};
\node[elim] at (0, -9.0)
    {\textit{Eliminates:} chance patterns in treatment effect heterogeneity};
\draw[arrow] (f4.south) -- (f5.north);

% Bottom: surviving explanation
\node[block=columbiadark, text=white, minimum width=7cm,
      minimum height=1.0cm, font=\normalsize\bfseries] (bottom) at (0, -12.0)
    {HOPE's eligibility--price channel};
\draw[arrow, very thick, columbiadark!70] (f5.south) -- (bottom.north);

% Challenge
\node[font=\footnotesize\itshape, text=gray!50!black, align=center] at (0, -13.2)
    {A confound must satisfy all five constraints simultaneously.};

\end{tikzpicture}
